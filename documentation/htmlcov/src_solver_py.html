13class Solver(object): 

14 """ 

15 The Solver performs stochastic gradient descent using different 

16 update rules defined in optim.py. 

17 

18 Solver needs both training and validation data to train and check validation accuracy 

19 Solver also needs a model, the neural net, and some configs 

20 

21 """

23 def __init__(self, model, data, **kwargs): 

24 """ 

25 Construct a new Solver instance. 

26 

27 Required arguments: 

28 - model: the neural net 

29 - data: A dictionary of training and validation data containing: 

30 

31 Optional arguments: 

32 - update_rule: A string giving the name of an update rule in optim.py. 

33 Default is 'sgd'. 

34 - optim_config: A dictionary containing hyperparameters that will be 

35 passed to the chosen update rule. Each update rule requires different 

36 hyperparameters (see optim.py) but all update rules require a 

37 'learning_rate' parameter so that should always be present. 

38 - lr_decay: A scalar for learning rate decay; after each epoch the 

39 learning rate is multiplied by this value. 

40 - batch_size: Size of minibatches used to compute loss and gradient 

41 during training. 

42 - num_epochs: The number of epochs to run for during training. 

43 - print_every: Integer; training losses will be printed every 

44 print_every iterations. 

45 - verbose: Boolean; if set to false then no output will be printed 

46 during training. 

47 - num_train_samples: Number of training samples used to check training 

48 accuracy; default is 1000; set to None to use entire training set. 

49 - num_val_samples: Number of validation samples to use to check val 

50 accuracy; default is None, which uses the entire validation set. 

51 - checkpoint_name: If not None, then save model checkpoints here every 

52 epoch. 

53 """ 

54 self.model = model 

55 self.X_train = data["X_train"] 

56 self.y_train = data["y_train"] 

57 self.X_val = data["X_val"] 

58 self.y_val = data["y_val"] 

59 

60 # Unpack keyword arguments 

61 self.update_rule = kwargs.pop("update_rule", "sgd") 

62 self.optim_config = kwargs.pop("optim_config", {}) 

63 self.lr_decay = kwargs.pop("lr_decay", 1.0) 

64 self.batch_size = kwargs.pop("batch_size", 100) 

65 self.num_epochs = kwargs.pop("num_epochs", 10) 

66 self.num_train_samples = kwargs.pop("num_train_samples", 1000) 

67 self.num_val_samples = kwargs.pop("num_val_samples", None) 

68 

69 self.checkpoint_name = kwargs.pop("checkpoint_name", None) 

70 self.print_every = kwargs.pop("print_every", 10) 

71 self.verbose = kwargs.pop("verbose", True) 

72 

73 # Throw an error if there are extra keyword arguments 

74 if len(kwargs) > 0: 

75 extra = ", ".join('"%s"' % k for k in list(kwargs.keys())) 

76 raise ValueError("Unrecognized arguments %s" % extra) 

77 

78 # Make sure the update rule exists, then replace the string 

79 # name with the actual function 

80 if not hasattr(optim, self.update_rule): 

81 raise ValueError('Invalid update_rule "%s"' % self.update_rule) 

82 self.update_rule = getattr(optim, self.update_rule) 

83 

84 self._reset()

86 def _reset(self): 

87 """ 

88 Reset the Solvers state at __init__ 

89 """ 

90 # Set up some variables for book-keeping 

91 self.epoch = 0 

92 self.best_val_acc = 0 

93 self.best_params = {} 

94 self.loss_history = [] 

95 self.train_acc_history = [] 

96 self.val_acc_history = [] 

97 

98 # Make a deep copy of the optim_config for each parameter 

99 self.optim_configs = {} 

100 for p in self.model.params: 

101 d = {k: v for k, v in self.optim_config.items()} 

102 self.optim_configs[p] = d

104 def _step(self): 

105 """ 

106 Make a single gradient update 

107 """ 

108 # Make a minibatch of training data 

109 num_train = self.X_train.shape[0] 

110 batch_mask = np.random.choice(num_train, self.batch_size) 

111 X_batch = self.X_train[batch_mask] 

112 y_batch = self.y_train[batch_mask] 

113 

114 # Compute loss and gradient 

115 loss, grads = self.model.loss(X_batch, y_batch) 

116 self.loss_history.append(loss) 

117 

118 # Perform a parameter update 

119 for p, w in self.model.params.items(): 

120 dw = grads[p] 

121 config = self.optim_configs[p] 

122 next_w, next_config = self.update_rule(w, dw, config) 

123 self.model.params[p] = next_w 

124 self.optim_configs[p] = next_config 

125

127 def check_accuracy(self, X, y, num_samples=None, batch_size=100): 

128 """ 

129 Check accuracy of the model on the provided data. 

130 

131 Inputs: 

132 - X: input data 

133 - y: labels 

134 - num_samples: If not None, subsample the data and only test the model 

135 on num_samples datapoints. 

136 - batch_size: Split X and y into batches of this size 

137 

138 Returns: 

139 - acc: fraction of instances that were correctly classified by the model. 

140 """ 

141 

142 # Maybe subsample the data 

143 N = X.shape[0] 

144 if num_samples is not None and N > num_samples: 

145 mask = np.random.choice(N, num_samples) 

146 N = num_samples 

147 X = X[mask] 

148 y = y[mask] 

149 

150 # Compute predictions in batches 

151 num_batches = N // batch_size 

152 if N % batch_size != 0: 

153 num_batches += 1 

154 y_pred = [] 

155 for i in range(num_batches): 

156 start = i * batch_size 

157 end = (i + 1) * batch_size 

158 scores = self.model.loss(X[start:end]) 

159 y_pred.append(np.argmax(scores, axis=1)) 

160 y_pred = np.hstack(y_pred) 

161 acc = np.mean(y_pred == y) 

162 

163 return acc

165 def train(self): 

166 """ 

167 Run optimization to train the model. 

168 """ 

169 num_train = self.X_train.shape[0] 

170 iterations_per_epoch = max(num_train // self.batch_size, 1) 

171 num_iterations = self.num_epochs * iterations_per_epoch 

172 

173 for t in range(num_iterations): 

174 self._step() 

175 

176 # Maybe print training loss 

177 if self.verbose and t % self.print_every == 0: 

178 print( 

179 "(Iteration %d / %d) loss: %f" 

180 % (t + 1, num_iterations, self.loss_history[-1]) 

181 ) 

182 

183 # At the end of every epoch, increment the epoch counter and decay 

184 # the learning rate. 

185 epoch_end = (t + 1) % iterations_per_epoch == 0 

186 if epoch_end: 

187 self.epoch += 1 

188 for k in self.optim_configs: 

189 self.optim_configs[k]["learning_rate"] *= self.lr_decay 

190 

191 # Check train and val accuracy on the first iteration, the last 

192 # iteration, and at the end of each epoch. 

193 first_it = t == 0 

194 last_it = t == num_iterations - 1 

195 if first_it or last_it or epoch_end: 

196 train_acc = self.check_accuracy( 

197 self.X_train, self.y_train, num_samples=self.num_train_samples 

198 ) 

199 val_acc = self.check_accuracy( 

200 self.X_val, self.y_val, num_samples=self.num_val_samples 

201 ) 

202 self.train_acc_history.append(train_acc) 

203 self.val_acc_history.append(val_acc) 

204 

205 if self.verbose: 

206 print( 

207 "(Epoch %d / %d) train acc: %f; val_acc: %f" 

208 % (self.epoch, self.num_epochs, train_acc, val_acc) 

209 ) 

210 

211 # Keep track of the best model 

212 if val_acc > self.best_val_acc: 

213 self.best_val_acc = val_acc 

214 self.best_params = {} 

215 for k, v in self.model.params.items(): 

216 self.best_params[k] = v.copy() 

217 

218 # At the end of training swap the best params into the model 

219 self.model.params = self.best_params
