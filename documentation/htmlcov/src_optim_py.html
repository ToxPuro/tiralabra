8def sgd(w, dw, config=None): 

9 """ 

10 Performs vanilla stochastic gradient descent. 

11 In order to decrease the loss function we need to go in the direction of negative gradient 

12 

13 config format: 

14 - learning_rate: Scalar learning rate. 

15 """ 

16 if config is None: 

17 config = {} 

18 config.setdefault("learning_rate", 1e-2) 

19 

20 w -= config["learning_rate"] * dw 

21 return w, config

23def sgd_momentum(w, dw, config=None): 

24 """ 

25 Performs stochastic gradient descent with momentum. 

26 With momentum added the neural net will keep 

27 on going with momentum so noise in the gradient doesn't 

28 affect as much and we don't get stuck in local minima 

29 

30 config format: 

31 - learning_rate: Scalar learning rate. 

32 - momentum: Scalar between 0 and 1 giving the momentum value. 

33 Setting momentum = 0 reduces to sgd. 

34 - velocity: A numpy array of the same shape as w and dw used to store a 

35 moving average of the gradients. 

36 """ 

37 if config is None: 

38 config = {} 

39 config.setdefault("learning_rate", 1e-2) 

40 config.setdefault("momentum", 0.9) 

41 v = config.get("velocity", np.zeros_like(w)) 

42 

43 v = config["momentum"] * v - config["learning_rate"] * dw 

44 next_w = w+v 

45 config["velocity"] = v 

46 

47 return next_w, config

50def rmsprop(w, dw, config=None): 

51 """ 

52 Uses the RMSProp update rule, which uses a moving average of squared 

53 gradient values to set adaptive per-parameter learning rates. 

54 read more from here: 

55 https://towardsdatascience.com/understanding-rmsprop-faster-neural-network-learning-62e116fcf29a 

56 

57 config format: 

58 - learning_rate: Scalar learning rate. 

59 - decay_rate: Scalar between 0 and 1 giving the decay rate for the squared 

60 gradient cache. 

61 - epsilon: Small scalar used for smoothing to avoid dividing by zero. 

62 - cache: Moving average of second moments of gradients. 

63 """ 

64 if config is None: 

65 config = {} 

66 config.setdefault("learning_rate", 1e-2) 

67 config.setdefault("decay_rate", 0.99) 

68 config.setdefault("epsilon", 1e-8) 

69 config.setdefault("cache", np.zeros_like(w)) 

70 

71 config["cache"] = config["decay_rate"]*config["cache"]+(1-config["decay_rate"])*dw**2 

72 next_w = w - config["learning_rate"]*dw /(np.sqrt(config["cache"]) + config["epsilon"]) 

73 

74 return next_w, config

77def adam(w, dw, config=None): 

78 """ 

79 Uses the Adam update rule, which incorporates moving averages of both the 

80 gradient and its square and a bias correction term. 

81 Read more from here 

82 https://machinelearningmastery.com/adam-optimization-algorithm-for-deep-learning/ 

83 

84 config format: 

85 - learning_rate: Scalar learning rate. 

86 - beta1: Decay rate for moving average of first moment of gradient. 

87 - beta2: Decay rate for moving average of second moment of gradient. 

88 - epsilon: Small scalar used for smoothing to avoid dividing by zero. 

89 - m: Moving average of gradient. 

90 - v: Moving average of squared gradient. 

91 - t: Iteration number. 

92 """ 

93 if config is None: 

94 config = {} 

95 config.setdefault("learning_rate", 1e-3) 

96 config.setdefault("beta1", 0.9) 

97 config.setdefault("beta2", 0.999) 

98 config.setdefault("epsilon", 1e-8) 

99 config.setdefault("m", np.zeros_like(w)) 

100 config.setdefault("v", np.zeros_like(w)) 

101 config.setdefault("t", 0) 

102 

103 eps, learning_rate = config['epsilon'], config['learning_rate'] 

104 beta1, beta2 = config['beta1'], config['beta2'] 

105 m, v, t = config['m'], config['v'], config['t'] 

106 # Adam 

107 t = t + 1 

108 m = beta1 * m + (1 - beta1) * dw # momentum 

109 mt = m / (1 - beta1**t) # bias correction 

110 v = beta2 * v + (1 - beta2) * (dw * dw) # RMSprop 

111 vt = v / (1 - beta2**t) # bias correction 

112 next_w = w - learning_rate * mt / (np.sqrt(vt) + eps) 

113 # update values 

114 config['m'], config['v'], config['t'] = m, v, t 

115 

116 return next_w, config
